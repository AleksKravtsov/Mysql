SELECT first_name, 
  (SELECT count(from_user_id) FROM messages WHERE messages.from_user_id = users.id ) as messag,
  (SELECT count(user_id) FROM posts where posts.user_id = users.id ) as post,
  (SELECT count(user_id) FROM media where media.user_id =users.id ) as media,
  (SELECT count(user_id) FROM likes where likes.user_id =users.id ) as lik
  FROM users;
